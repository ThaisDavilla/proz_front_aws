Algoritmo "PessoasEntrevistadas"
var
  contador, quantidade: inteiro

 

inicio
  contador <- 1
  escreva("Quantas pessoas você entrevistou hoje? ")
  leia(quantidade)

 

  enquanto (contador <= quantidade) faca
    escreva("Você entrevistou a pessoa de código 00", contador)

  fimenquanto
